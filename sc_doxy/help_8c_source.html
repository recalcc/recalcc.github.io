00540 void
00541 help()
00542 {
00543     int option;
00544     char **ns = intro;
00545 
00546     while((option = pscreen(ns)) != 'q' && option != 'Q') {
00547         switch (option) {
00548         case 'a': case 'A': ns = intro; break;
00549         case 'b': case 'B': ns = toggleoptions; break;
00550         case 'c': case 'C': ns = setoptions; break;
00551         case 'd': case 'D': ns = cursor; break;
00552         case 'e': case 'E': ns = cell; break;
00553         case 'f': case 'F': ns = vi; break;
00554         case 'g': case 'G': ns = file; break;
00555         case 'h': case 'H': ns = row; break;
00556         case 'i': case 'I': ns = range; break;
00557         case 'j': case 'J': ns = misc; break;
00558         case 'k': case 'K': ns = var; break;
00559         case 'l': case 'L': ns = rangef; break;
00560         case 'm': case 'M': ns = numericf; break;
00561         case 'n': case 'N': ns = stringf; break;
00562         case 'o': case 'O': ns = finf; break;
00563         case 'p': case 'P': ns = timef; break;
00564         default: ns = intro; break;
00565         }
00566     }
00567     FullUpdate++;
00568     (void) move(1,0);
00569     (void) clrtobot();
00570 }
